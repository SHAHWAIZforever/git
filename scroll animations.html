// For mobile touch events window.addEventListener('touchstart', handleScroll, false); window.addEventListener('touchmove', handleScroll, false); window.addEventListener('touchend', handleScroll, false); // Scroll event for desktop window.addEventListener('scroll', handleScroll, false); function handleScroll() { const fadeElements = document.querySelectorAll('.fade-up, .fade-up-1, .fade-up-2, .fade-up-3, .fade-left, .fade-right, .fade-down, .fade-up-slow'); fadeElements.forEach(function (element) { const rect = element.getBoundingClientRect(); if (rect.top <= window.innerHeight && rect.bottom >= 0) { element.classList.add('visible'); } else { element.classList.remove('visible'); } }); } /* General fade-up animation */ .fade-up { opacity: 0; transform: translateY(150px); /* Start position (a bit further down) */ transition: opacity 0.5s ease, transform 1s ease; } .fade-up.visible { opacity: 1; transform: translateY(0); /* Final position */ } /* General fade-down animation */ .fade-down { opacity: 0; transform: translateY(-60px) scale(0.8); /* Start higher and smaller */ transition: opacity 1s ease, transform 0.5s ease; } .fade-down.visible { opacity: 1; transform: translateY(0) scale(1); /* Final position and size */ } /* Fade-up slow animation */ .fade-up-slow { opacity: 0; transform: translateY(150px); /* Start position (below) */ transition: opacity 1s ease 0.3s, transform 2s ease 0.3s; /* 1 second duration with 0.3s delay */ } .fade-up-slow.visible { opacity: 1; transform: translateY(0); /* Final position */ } /* Fade-left animation */ .fade-left { opacity: 0; transform: translateX(-200px); /* Start position (left) */ transition: opacity 1s ease, transform 1s ease; } .fade-left.visible { opacity: 1; transform: translateX(0); /* Final position */ } /* Fade-right animation */ .fade-right { opacity: 0; transform: translateX(200px); /* Start position (right) */ transition: opacity 1s ease, transform 1s ease; } .fade-right.visible { opacity: 1; transform: translateX(0); /* Final position */ } /* Fade-up 1 (faster) */ .fade-up-1 { opacity: 0; transform: translateY(50px); /* Start position (below) */ transition: opacity 0.3s ease, transform 0.3s ease; /* Faster */ } .fade-up-1.visible { opacity: 1; transform: translateY(0); /* Final position */ } /* Fade-up 2 (slightly slower with delay) */ .fade-up-2 { opacity: 0; transform: translateY(50px); /* Start position */ transition: opacity 0.3s ease 0.3s, transform 0.3s ease 0.3s; /* Faster with short delay */ } .fade-up-2.visible { opacity: 1; transform: translateY(0); /* Final position */ } /* Fade-up 3 (slower than fade-up 2) */ .fade-up-3 { opacity: 0; transform: translateY(50px); /* Start position */ transition: opacity 0.3s ease 0.6s, transform 0.3s ease 0.6s; /* Faster with delay after fade-up 2 */ } .fade-up-3.visible { opacity: 1; transform: translateY(0); /* Final position */ }
